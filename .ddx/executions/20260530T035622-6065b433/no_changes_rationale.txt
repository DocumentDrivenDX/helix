  #!/bin/bash
  # Verify the integration test infrastructure is complete and well-formed

  set -euo pipefail

  REPO_ROOT=$(pwd)
  GENIE_DIR="$REPO_ROOT/tests/workflows/genie"

  # Check directory structure
  [[ -d "$GENIE_DIR/scenarios" ]] || exit 1
  [[ -d "$GENIE_DIR/expected" ]] || exit 1
  [[ -d "$GENIE_DIR/evals" ]] || exit 1
  [[ -d "$GENIE_DIR/recordings" ]] || exit 1

  # Check scenario prompt files
  [[ -f "$GENIE_DIR/scenarios/skill-list.prompt" ]] || exit 1
  [[ -f "$GENIE_DIR/scenarios/install-verify.prompt" ]] || exit 1
  [[ -f "$GENIE_DIR/scenarios/bootstrap.prompt" ]] || exit 1

  # Check expected output files
  [[ -f "$GENIE_DIR/expected/skill-list.expect" ]] || exit 1
  [[ -f "$GENIE_DIR/expected/install-verify.expect" ]] || exit 1
  [[ -f "$GENIE_DIR/expected/bootstrap.expect" ]] || exit 1

  # Check evaluation rules
  [[ -f "$GENIE_DIR/evals/routing.jsonl" ]] || exit 1

  # Validate evaluation rules are well-formed JSONL
  while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    echo "$line" | jq . > /dev/null 2>&1 || exit 1
  done < "$GENIE_DIR/evals/routing.jsonl"

  # Check main driver script
  [[ -f "$GENIE_DIR/run-scenarios.sh" ]] || exit 1
  [[ -x "$GENIE_DIR/run-scenarios.sh" ]] || exit 1

  # Check README
  [[ -f "$GENIE_DIR/README.md" ]] || exit 1

  # Check docs/install/databricks-genie.md has been updated to reference the test
  grep -q "tests/workflows/genie" "$REPO_ROOT/docs/install/databricks-genie.md" || exit 1
  grep -q "workflow integration test" "$REPO_ROOT/docs/install/databricks-genie.md" || exit 1

  echo "✓ All infrastructure files present and well-formed"
  exit 0
